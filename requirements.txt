Debian/Ubuntu OS based installation, with root user:

-- Packages installation

sudo mkdir -p /etc/apt/keyrings
curl -fsSL https://repo.charm.sh/apt/gpg.key | sudo gpg --yes --dearmor -o /etc/apt/keyrings/charm.gpg
echo "deb [signed-by=/etc/apt/keyrings/charm.gpg] https://repo.charm.sh/apt/ * *" | sudo tee /etc/apt/sources.list.d/charm.list

apt-get update
apt-get install -y build-essential git curl wget gpg tar libpcre3-dev libssl-dev libpcap-dev net-tools xsltproc bind9-dnsutils toilet boxes lolcat gum netcat-traditional

Verification:
xsltproc -V
ipcalc -v

-- Masscan installation

cd /tmp
git clone https://github.com/robertdavidgraham/masscan.git
cd masscan
make -j$(nproc)
mv bin/masscan /usr/bin/
updatedb

Verification:
masscan -V

-- Nmap installation

cd /tmp
wget https://nmap.org/dist/nmap-7.95.tgz
tar -xvzf nmap-7.95.tgz
cd nmap-7.95
./configure
make -j$(nproc)
make install
updatedb

Verification:
nmap -V

-- Vulners.nse Nmap Script installation

cd /tmp
git clone https://github.com/vulnersCom/nmap-vulners
cp nmap-vulners/vulners.nse /usr/local/share/nmap/scripts/
updatedb
nmap --script-updatedb

Verification:
nmap -sV --script vulners <target>
